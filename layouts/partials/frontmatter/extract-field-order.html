{{/* ============================================================================= Extract Field Order Partial ============================================================================= Extracts the order of fields from raw YAML config text by parsing indentation and tracking the hierarchy. Returns a dict mapping field paths to order indices. This is necessary because Go maps don't preserve insertion order, so we need to parse the raw config file to determine the original order of fields. Handles both list format (for top-level fields) and map format (for nested fields): - List format: `- key: fieldName` followed by properties - Map format: `fieldName:` followed by properties Input (dict): - configContent: Raw config file content as string Output: Dict mapping field paths to order indices Example: { "status": 0, "slug": 1, "object input": 2, "object input.fields.author": 0, "object input.fields.role": 1, ... } Usage: {{ $fieldOrders := partial "frontmatter/extract-field-order" (dict "configContent" $configContent) }} ============================================================================= */}} {{- $configContent := .configContent -}} {{- $lines := split $configContent "\n" -}} {{- $orders := dict -}} {{/* Track our position in the hierarchy */}} {{- $inFrontmatter := false -}} {{- $frontmatterIndent := -1 -}} {{/* Track indent levels and their corresponding paths and order counters */}} {{- $indentPaths := dict -}} {{- $indentOrders := dict -}} {{/* Track if we're inside a list item's properties (to skip them) */}} {{- $listItemIndent := -1 -}} {{- $listItemPath := "" -}} {{/* Track fields: keyword indent to know when we're in a fields block */}} {{- $fieldsKeywordIndent := -1 -}} {{- range $lineNum, $line := $lines -}} {{- $trimmed := strings.TrimLeft " " $line -}} {{- $currentIndent := sub (len $line) (len $trimmed) -}} {{/* Skip empty lines and comments */}} {{- if and (gt (len $trimmed) 0) (not (hasPrefix $trimmed "#")) -}} {{/* Detect when we enter frontmatter section */}} {{- if and (not $inFrontmatter) (findRE "^frontmatter:\\s*$" $trimmed) -}} {{- $inFrontmatter = true -}} {{- $frontmatterIndent = $currentIndent -}} {{- $indentPaths = merge $indentPaths (dict (string $currentIndent) "") -}} {{- $indentOrders = merge $indentOrders (dict (string $currentIndent) 0) -}} {{- else if $inFrontmatter -}} {{/* Check if we've exited the frontmatter section */}} {{- if le $currentIndent $frontmatterIndent -}} {{- $inFrontmatter = false -}} {{- else -}} {{/* Check if we've exited a list item's property block */}} {{- if and (ge $listItemIndent 0) (le $currentIndent $listItemIndent) -}} {{- $listItemIndent = -1 -}} {{- $listItemPath = "" -}} {{- $fieldsKeywordIndent = -1 -}} {{- end -}} {{/* Check if we've exited a fields: block */}} {{- if and (ge $fieldsKeywordIndent 0) (le $currentIndent $fieldsKeywordIndent) -}} {{- $fieldsKeywordIndent = -1 -}} {{- end -}} {{/* Process this line */}} {{- $fieldName := "" -}} {{- $isFieldsKeyword := false -}} {{- $isFieldKeyword := false -}} {{- $isListItem := false -}} {{/* Check for special keywords and field patterns */}} {{/* Note: fields: and field: must be checked first, even inside list item properties */}} {{- if findRE "^fields:\\s*$" $trimmed -}} {{- $isFieldsKeyword = true -}} {{- else if findRE "^field:\\s*$" $trimmed -}} {{- $isFieldKeyword = true -}} {{- else if findRE "^- key:" $trimmed -}} {{/* List format entry: `- key: fieldName` */}} {{- $isListItem = true -}} {{- $fieldName = replaceRE "^- key:\\s*(.+?)\\s*$" "$1" $trimmed -}} {{- else if and (ge $listItemIndent 0) (gt $currentIndent $listItemIndent) (lt $fieldsKeywordIndent 0) -}} {{/* Inside a list item's properties but NOT in a fields: block - skip regular properties */}} {{/* (fields:/field: already handled above, so only skip other properties) */}} {{- else if and (not (hasPrefix $trimmed "-")) (findRE "^[a-zA-Z0-9_ -]+:" $trimmed) -}} {{/* Map format entry: `fieldName:` (only if not inside list item properties) */}} {{- $fieldName = replaceRE "^([^:]+):.*$" "$1" $trimmed -}} {{- end -}} {{/* Find the parent indent level */}} {{- $parentIndent := sub $currentIndent 2 -}} {{- $parentPath := index $indentPaths (string $parentIndent) | default "" -}} {{/* For list items, adjust parent to be the frontmatter level */}} {{- if $isListItem -}} {{- $parentIndent = $frontmatterIndent -}} {{- $parentPath = index $indentPaths (string $parentIndent) | default "" -}} {{- end -}} {{- if $isFieldsKeyword -}} {{/* 'fields:' keyword - update path for children */}} {{/* Use parentPath from indentPaths, fallback to listItemPath for direct children of list items */}} {{- $basePath := $parentPath -}} {{- if and (not $basePath) $listItemPath -}} {{- $basePath = $listItemPath -}} {{- end -}} {{- $newPath := printf "%s.fields" $basePath -}} {{- $indentPaths = merge $indentPaths (dict (string $currentIndent) $newPath) -}} {{- $indentOrders = merge $indentOrders (dict (string $currentIndent) 0) -}} {{/* Mark that we're in a fields: block */}} {{- $fieldsKeywordIndent = $currentIndent -}} {{- else if $isFieldKeyword -}} {{/* 'field:' keyword - update path for children */}} {{/* Use parentPath from indentPaths, fallback to listItemPath for direct children of list items */}} {{- $basePath := $parentPath -}} {{- if and (not $basePath) $listItemPath -}} {{- $basePath = $listItemPath -}} {{- end -}} {{- $newPath := printf "%s.field" $basePath -}} {{- $indentPaths = merge $indentPaths (dict (string $currentIndent) $newPath) -}} {{- $indentOrders = merge $indentOrders (dict (string $currentIndent) 0) -}} {{/* Mark that we're in a field: block */}} {{- $fieldsKeywordIndent = $currentIndent -}} {{- else if $fieldName -}} {{/* This is a field - record its order */}} {{- $fieldNameLower := lower $fieldName -}} {{- $fullPath := $fieldNameLower -}} {{- if $parentPath -}} {{- $fullPath = printf "%s.%s" $parentPath $fieldNameLower -}} {{- end -}} {{/* Get and increment the order counter for this level */}} {{- $currentOrder := index $indentOrders (string $parentIndent) | default 0 -}} {{- $orders = merge $orders (dict $fullPath $currentOrder) -}} {{- $indentOrders = merge $indentOrders (dict (string $parentIndent) (add $currentOrder 1)) -}} {{/* Set this field as the path for its indent level */}} {{- $indentPaths = merge $indentPaths (dict (string $currentIndent) $fullPath) -}} {{- $indentOrders = merge $indentOrders (dict (string $currentIndent) 0) -}} {{/* If this is a list item, track that we're in its properties */}} {{- if $isListItem -}} {{- $listItemIndent = $currentIndent -}} {{- $listItemPath = $fullPath -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- return $orders -}}
